    def initialize(*args)
      case args.size
      when 0
        @center = PgPoint.new
        @radius = 0
      when 2
        @center = args[0]
        @radius = args[1]
      when 3
        @center = PgPoint.new(*args[0..1])
        @radius = args[2]
      else
        raise ArgumentError, "Incorrect number of arguments: #{args.size}"
      end
    end

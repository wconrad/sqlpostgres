      def from_sql(s)
        if s =~ /^(\[)\(.*\)(,\(.*\))?\]$/ || s =~ /^(\()\(.*\)(,\(.*\))?\)$/
          closed = $1 == "("
          points = s.scan(/\([^(]*?\)/).collect do |t|
            PgPoint.from_sql(t)
          end
          PgPath.new(closed, *points)
        else
          raise ArgumentError, "Invalid path: #{s.inspect}"
        end
      end

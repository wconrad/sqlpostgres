      def from_sql(s)
        if s =~ /^(\d+):(\d+):(\d+)((?:\+|-)\d+)(?::(\d+))?$/
          PgTimeWithTimeZone.new($1.to_i, $2.to_i, $3.to_i, $4.to_i, $5.to_i)
        else
          raise ArgumentError, "Invalid time with time zone: #{s.inspect}"
        end
      end

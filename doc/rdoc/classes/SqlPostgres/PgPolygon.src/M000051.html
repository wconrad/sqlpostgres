      def from_sql(s)
        if s =~ /^(\()\(.*\)(,\(.*\))?\)$/
          points = s.scan(/\([^(]*?\)/).collect do |t|
            PgPoint.from_sql(t)
          end
          PgPolygon.new(*points)
        else
          raise ArgumentError, "Invalid polygon: #{s.inspect}"
        end
      end

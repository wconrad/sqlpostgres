    def initialize(name, connection = Connection.default)
      @name = name
      @state = :open
      @finished = false
      @connection = connection
      @connection.exec("savepoint #{name}")
      begin
        result = yield(self)
        commit
        result
      rescue Exception
        abort
        raise
      end
    end

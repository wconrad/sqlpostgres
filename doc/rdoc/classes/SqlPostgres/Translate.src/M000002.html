    def substitute_values(expression)
      if expression.is_a?(Array)
        pieces = expression[1..-1].collect do |value|
          escape_sql(value)
        end
        if expression.first == :in
          "(#{pieces.join(', ')})"
        else
          expression[0] % pieces
        end
      else
        expression
      end
    end

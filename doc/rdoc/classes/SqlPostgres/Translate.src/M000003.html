    def escape_array(a)
      if a.is_a?(Array)
        if a.empty?
          "'{}'"
        else
          pieces = a.collect do |e|
            escape_array(e)
          end
          "ARRAY[#{pieces.join(', ')}]"
        end
      else
        escape_sql(a)
      end
    end
